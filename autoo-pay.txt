package com.creditone.ucrm.autopaysynch.clienthandler;

import java.util.Map;
import java.util.function.Function;

import org.springframework.http.HttpStatusCode;
import org.springframework.http.MediaType;
import org.springframework.stereotype.Component;
import org.springframework.util.CollectionUtils;
import org.springframework.util.MultiValueMap;
import org.springframework.web.reactive.function.client.ClientResponse;
import org.springframework.web.reactive.function.client.WebClient;
import org.springframework.web.util.UriComponentsBuilder;

import com.creditone.ucrm.autopaysynch.constant.AutoPaySynchConstant;
import com.creditone.ucrm.autopaysynch.exception.AutoPaySynchException;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import reactor.core.publisher.Mono;

/**
 * Client handler component responsible for making HTTP GET requests to external services. This
 * class provides a centralized way to handle web client operations with proper error handling,
 * header management, and query parameter processing for autopay synchronization operations.
 */
@Slf4j
@Component
@RequiredArgsConstructor
public class AutoPaySyncClientHandler {

	private final WebClient autoPayWebClient;

	/**
	 * Executes an HTTP GET request to the specified URL with optional headers and query parameters.
	 * This method constructs the complete request URL, applies headers, and configures error handling
	 * for the web client call.
	 *
	 * @param url
	 * @param headers
	 * @param targetApplication
	 * @param params
	 * @return
	 */
	public WebClient.ResponseSpec get(String url, Map<String, String> headers, String targetApplication, MultiValueMap<String, String> params) {
		String finalUrl = url;
		if (params != null && !params.isEmpty()) {
			finalUrl = UriComponentsBuilder.fromUriString(url).queryParams(params).build().toString();
		}

		var requestHeaderSpec = autoPayWebClient.get().uri(finalUrl);

		String correlationId = "";
		if (!CollectionUtils.isEmpty(headers)) {
			correlationId = (String) headers.get(AutoPaySynchConstant.CORRELATION_ID);
			for (Map.Entry<String, String> headerEntry : headers.entrySet()) {
				requestHeaderSpec.header(headerEntry.getKey(), headerEntry.getValue());
			}
		}

		return requestHeaderSpec.accept(MediaType.APPLICATION_JSON).retrieve().onStatus(HttpStatusCode::isError, errorResponseMappingFunction(targetApplication, correlationId));
	}

	/**
	 * Creates a function that handles HTTP error responses by mapping them to custom exceptions. This
	 * method provides standardized error handling across all web client calls by extracting error
	 * response bodies and converting them into meaningful exception messages.
	 *
	 * @param targetApplication
	 * @param correlationId
	 * @return
	 */
	private Function<ClientResponse, Mono<? extends Throwable>> errorResponseMappingFunction(String targetApplication, String correlationId) {
		return errorResponse -> errorResponse.createException().doOnNext(ex -> log.error(AutoPaySynchConstant.LOG_PREFIX
				+ "External service call failed [app={}, status={}, url={}]: {}", correlationId, targetApplication, ex.getStatusCode().value(), ex.getRequest() != null
						? ex.getRequest().getURI()
						: "N/A", ex.getResponseBodyAsString())).flatMap(ex -> Mono.error(new AutoPaySynchException("External service call failed for targetApp: "
								+ targetApplication + " with status: " + ex.getStatusCode().value() + " and response: " + ex.getResponseBodyAsString(), correlationId, ex)));
	}
}

		try {
			ConnectionFactoryOptions options = ConnectionFactoryOptions.builder().option(DRIVER, "postgresql").option(HOST, databaseConfig.getHost()).option(PORT, databaseConfig

			connectionPool = new ConnectionPool(poolConfiguration);

			log.error("Failed to initialize AutoPay database connection factory Error: {} Cause: {}", e.getMessage(), ExceptionUtils.getStackTrace(e).replaceAll("\\r?\\n", " "));
			throw new IllegalStateException("Database connection initialization failed", e);
		}
	}

		log.info("Creating AutoPay R2DBC entity template");

	public ReactiveTransactionManager autopayTransactionManager(@Qualifier("autopayConnectionFactory") ConnectionFactory connectionFactory) {

		log.info("Creating AutoPay transaction manager");
		return new R2dbcTransactionManager(connectionFactory);
	}

	/** Graceful shutdown of connection pool */
	@PreDestroy
	public void cleanup() {
		log.info("Shutting down AutoPay database connection pool");
		try {
			if (connectionPool != null && !connectionPool.isDisposed()) {
				connectionPool.dispose();
				log.info("AutoPay database connection pool disposed successfully");
			}
		} catch (Exception e) {
			log.error("Error during AutoPay database connection pool cleanup Error: {}, Cause: {}", e.getMessage(), ExceptionUtils.getStackTrace(e).replaceAll("\\r?\\n", " "));

package com.creditone.ucrm.autopaysynch.config;
import java.util.Properties;
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.boot.context.properties.EnableConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.apache.commons.lang3.exception.ExceptionUtils;
import com.hazelcast.client.HazelcastClient;
import com.hazelcast.client.config.ClientConfig;
import com.hazelcast.client.impl.connection.tcp.RoutingMode;
import com.hazelcast.config.SSLConfig;
import com.hazelcast.core.HazelcastInstance;
import lombok.Data;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
/** Configuration class for Hazelcast client setup in the autopay synchronization service. */
@Slf4j
@Configuration
@RequiredArgsConstructor
@EnableConfigurationProperties(HazelcastConfiguration.HazelcastConfigProperties.class)
public class HazelcastConfiguration {

    private final HazelcastConfigProperties properties;
    /** Configuration properties for Hazelcast client connection settings. */
    @Data
    @ConfigurationProperties(prefix = "hazelcast")
    public static class HazelcastConfigProperties {
        private String clusterName;
        private String address;
        private String trustStorePath;
        private String trustStorePassword;
        private String username;
        private String password;
        private Boolean enabled;
    }
    /**
     * Creates and configures a Hazelcast client instance as a Spring bean.
     *
     * @return HazelcastInstance
     */
    @Bean
    @ConditionalOnProperty(name = "hazelcast.enabled", havingValue = "true", matchIfMissing = true)
    public HazelcastInstance hazelcastInstance() {
        if (!properties.getEnabled()) {
            log.info("Hazelcast is disabled via configuration");
            return null;
        }
        log.info("Configuring Hazelcast client instance...");
        try {
            ClientConfig clientConfig = buildClientConfig();
            HazelcastInstance instance = HazelcastClient.newHazelcastClient(clientConfig);
            log.info("Hazelcast client instance configured successfully.");
            return instance;
        } catch (Exception e) {
            log.warn("Error instantiating Hazelcast, error: {} cause: {}", e.getMessage(), ExceptionUtils.getStackTrace(e).replaceAll("\\r?\\n", " "));
        }
        return null;
    }
    /**
     * Builds the Hazelcast client configuration with security and network settings.
     *
     * @return ClientConfig
     */
    private ClientConfig buildClientConfig() {
        ClientConfig clientConfig = new ClientConfig();
        clientConfig.setClusterName(properties.getClusterName());
        // Network configuration
        clientConfig.getNetworkConfig().addAddress(properties.getAddress()).getClusterRoutingConfig().setRoutingMode(RoutingMode.ALL_MEMBERS);
        // SSL configuration
        log.info("Setting up hazelcast SSL configuration.");
        Properties sslProps = new Properties();
        sslProps.setProperty("trustStore", properties.getTrustStorePath());
        sslProps.setProperty("trustStorePassword", properties.getTrustStorePassword());
        clientConfig.getNetworkConfig().setSSLConfig(new SSLConfig().setEnabled(true).setProperties(sslProps));
        // Security configuration
        log.info("Setting up hazelcast security credentials.");
        clientConfig.getSecurityConfig().setUsernamePasswordIdentityConfig(properties.getUsername(), properties.getPassword());
        return clientConfig;
    }

}


package com.creditone.ucrm.autopaysynch.config;
import java.util.Optional;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.actuate.health.Health;
import org.springframework.boot.actuate.health.HealthIndicator;
import org.springframework.stereotype.Component;
import org.apache.commons.lang3.exception.ExceptionUtils;
import com.hazelcast.core.HazelcastInstance;
import lombok.extern.slf4j.Slf4j;
@Slf4j
@Component
public class HazelcastHealthIndicator implements HealthIndicator {
    private final Optional<HazelcastInstance> hazelcastInstance;
    public HazelcastHealthIndicator(@Autowired(required = false) HazelcastInstance hazelcastInstance) {
        this.hazelcastInstance = Optional.ofNullable(hazelcastInstance);
        log.info("HazelcastHealthIndicator initialized with instance present: {}", this.hazelcastInstance.isPresent());
    }
    @Override
    public Health health() {
        try {
            if (hazelcastInstance.isEmpty()) {
                log.warn("Hazelcast instance not available - reporting UNKNOWN status");
                return Health.unknown().withDetail("status", "Hazelcast not configured or disabled").withDetail("reason", "No HazelcastInstance bean found").build();
            }
            HazelcastInstance instance = hazelcastInstance.get();
            // Check if Hazelcast is running
            if (instance.getLifecycleService().isRunning()) {
                log.debug("Hazelcast instance is running - reporting UP status");
                return Health.up().withDetail("name", instance.getName()).withDetail("memberCount", instance.getCluster().getMembers().size()).withDetail("status", "Running")
                        .build();
            } else {
                log.warn("Hazelcast instance exists but is not running - reporting DOWN status");
                return Health.down().withDetail("name", instance.getName()).withDetail("status", "Not running").withDetail("reason",
                        "Lifecycle service indicates instance is not running").build();
            }
        } catch (Exception e) {
            log.error("Error checking Hazelcast health: {} cause: {}", e.getMessage(), ExceptionUtils.getStackTrace(e).replaceAll("\\r?\\n", " "));
            return Health.down().withException(e).withDetail("error", e.getMessage()).withDetail("status", "Health check failed").build();
        }
    }
}

/**
 * Configuration class responsible for creating and configuring reactive WebClient instances for
 * debit payment synchronization HTTP operations.
 */
@Slf4j
@Configuration
@RefreshScope
@RequiredArgsConstructor
@EnableConfigurationProperties(WebClientConfiguration.HttpConfigProperties.class)
public class WebClientConfiguration {

	private final HttpConfigProperties httpConfig;

	/**
	 * Configuration properties class that encapsulates all HTTP client settings. These properties are
	 * externalized to allow environment-specific tuning without code changes.
	 */
	@Data
	@ConfigurationProperties(prefix = "config.http")
	public static class HttpConfigProperties {
		private Integer maxConnections;
		private Integer retryCount;
		private Integer retryDurationSeconds;
		private Integer maxInmemorySize;
		private Integer maxConnectionPool;
		private Integer maxIdleTimeSeconds;
		private Integer maxLifeTimeSeconds;
		private Integer evictInBackgroundSeconds;
		private Integer connectionTimeoutMillis;
		private Integer responseTimeoutMillis;
		private Integer readTimeoutMillis;
		private Integer writeTimeoutMillis;
		private Boolean enableMetrics;
	}

	/**
	 * Creates a specialized ObjectMapper for payment-related JSON operations. This mapper is
	 * configured for robust deserialization that can handle varying JSON structures from different
	 * payment systems.
	 *
	 * @return
	 */
	@Bean("paymentsObjectMapper")
	public ObjectMapper paymentsObjectMapper() {
		return new ObjectMapper().setSerializationInclusion(JsonInclude.Include.NON_NULL).disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES).findAndRegisterModules();
	}

	/**
	 * Creates the primary WebClient instance for payment synchronization operations. This client is
	 * specifically configured for debit payment service interactions with optimized settings for
	 * reliability and performance.
	 *
	 * @param paymentsObjectMapper
	 * @return
	 */
	@Bean("paymentsWebClient")
	@DependsOn("paymentsObjectMapper")
	public WebClient paymentsWebClient(ObjectMapper paymentsObjectMapper) {
		return createWebClient(paymentsObjectMapper, "debit-payments-synch");
	}

	/**
	 * Factory method that creates a fully configured WebClient instance with enterprise-grade
	 * settings for production use.
	 *
	 * @param objectMapper
	 * @param clientName
	 * @return
	 */
	private WebClient createWebClient(ObjectMapper objectMapper, String clientName) {
		ConnectionProvider provider = createConnectionProvider(clientName);
		HttpClient httpClient = createHttpClient(provider);
		ExchangeStrategies exchangeStrategies = createExchangeStrategies(objectMapper);

		return WebClient.builder().clientConnector(new ReactorClientHttpConnector(httpClient)).filter(buildRetryExchangeFilterFunction()).exchangeStrategies(exchangeStrategies)
				.defaultHeaders(this::setDefaultHeaders).build();
	}

	/**
	 * Creates a connection provider with advanced connection pool management. This provider handles
	 * connection lifecycle, resource optimization, and background maintenance for optimal
	 * performance.
	 *
	 * @param name
	 * @return
	 */
	private ConnectionProvider createConnectionProvider(String name) {
		ConnectionProvider.Builder builder = ConnectionProvider.builder(name + "-connection-pool").maxConnections(httpConfig.getMaxConnections()).maxIdleTime(Duration.ofSeconds(
				httpConfig.getMaxIdleTimeSeconds())).maxLifeTime(Duration.ofSeconds(httpConfig.getMaxLifeTimeSeconds())).pendingAcquireMaxCount(httpConfig.getMaxConnectionPool())
				.evictInBackground(Duration.ofSeconds(httpConfig.getEvictInBackgroundSeconds()));

		if (httpConfig.getEnableMetrics()) {
			builder.metrics(true);
		}

		return builder.build();
	}

	/**
	 * Creates an HttpClient with comprehensive timeout handling and optional metrics. This client
	 * provides multiple layers of timeout protection to handle various failure scenarios and network
	 * conditions.
	 *
	 * @param provider
	 * @return
	 */
	private HttpClient createHttpClient(ConnectionProvider provider) {
		HttpClient httpClient = HttpClient.create(provider).option(ChannelOption.CONNECT_TIMEOUT_MILLIS, httpConfig.getConnectionTimeoutMillis()).responseTimeout(Duration.ofMillis(
				httpConfig.getResponseTimeoutMillis())).doOnConnected(conn -> conn.addHandlerFirst(new ReadTimeoutHandler(httpConfig.getReadTimeoutMillis(), TimeUnit.MILLISECONDS))
						.addHandlerLast(new WriteTimeoutHandler(httpConfig.getWriteTimeoutMillis(), TimeUnit.MILLISECONDS)));

		if (httpConfig.getEnableMetrics()) {
			httpClient = httpClient.metrics(true, Function.identity());
		}

		return httpClient;
	}

	/**
	 * Creates exchange strategies for JSON processing with memory management. These strategies define
	 * how HTTP request/response bodies are serialized and deserialized, with controls for memory
	 * usage.
	 *
	 * @param objectMapper
	 * @return
	 */
	private ExchangeStrategies createExchangeStrategies(ObjectMapper objectMapper) {
		return ExchangeStrategies.builder().codecs(configurer -> {
			configurer.defaultCodecs().maxInMemorySize(httpConfig.getMaxInmemorySize());
			configurer.defaultCodecs().jackson2JsonDecoder(new Jackson2JsonDecoder(objectMapper, MediaType.APPLICATION_JSON));
			configurer.defaultCodecs().jackson2JsonEncoder(new Jackson2JsonEncoder(objectMapper, MediaType.APPLICATION_JSON));
		}).build();
	}

	/**
	 * Creates an exchange filter function that implements intelligent retry logic with exponential
	 * backoff for handling transient failures.
	 *
	 * @return
	 */
	private ExchangeFilterFunction buildRetryExchangeFilterFunction() {
		return (request, next) -> next.exchange(request).retryWhen(Retry.backoff(httpConfig.getRetryCount(), Duration.ofSeconds(httpConfig.getRetryDurationSeconds())).filter(
				this::isRetryableException).doBeforeRetry(retrySignal -> log.warn("Retrying request to {} due to: {} (attempt: {})", request.url(), retrySignal.failure()
						.getMessage(), retrySignal.totalRetries() + 1)).onRetryExhaustedThrow((retryBackoffSpec, retrySignal) -> new ServiceUnavailableException(String.format(
								"Failed to process request to %s after %d retries. Last error:" + " %s", request.url(), httpConfig.getRetryCount(), retrySignal.failure()
										.getMessage()), retrySignal.failure())));
	}

	/**
	 * Determines whether an exception should trigger a retry attempt.
	 *
	 * @param throwable
	 * @return
	 */
	private boolean isRetryableException(Throwable throwable) {
		return throwable instanceof ConnectException || throwable instanceof IOException || throwable instanceof java.net.SocketTimeoutException
				|| throwable instanceof java.nio.channels.ClosedChannelException || (throwable instanceof WebClientResponseException wcre && isRetryableStatus(wcre
						.getStatusCode()));
	}

	private boolean isRetryableStatus(HttpStatusCode statusCode) {
		return statusCode.is5xxServerError() || statusCode == HttpStatus.REQUEST_TIMEOUT || statusCode == HttpStatus.TOO_MANY_REQUESTS;
	}

	private void setDefaultHeaders(HttpHeaders headers) {
		headers.set(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_JSON_VALUE);
		headers.set(HttpHeaders.ACCEPT, MediaType.APPLICATION_JSON_VALUE);
		headers.set(HttpHeaders.USER_AGENT, "DebitPaymentSyncService/1.0");
	}
}

package com.creditone.ucrm.autopaysynch.filter;

import java.time.ZonedDateTime;
import java.util.Arrays;
import java.util.HashSet;
import java.util.Set;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;
import org.springframework.util.CollectionUtils;
import org.springframework.util.ObjectUtils;
import org.springframework.web.server.ServerWebExchange;
import org.springframework.web.server.WebFilter;
import org.springframework.web.server.WebFilterChain;

import com.creditone.microservice.util.UUIDGenerator;
import com.creditone.ucrm.autopaysynch.constant.AutoPaySynchConstant;

import lombok.extern.slf4j.Slf4j;
import reactor.core.publisher.Mono;
import reactor.util.context.Context;

@Component
@Slf4j
public class AutoPaySyncFilter implements WebFilter {
	private String appName;
	private static Set<String> uriSubList = new HashSet<String>();

	private String healthUri;

	public AutoPaySyncFilter(@Value(value = "${APP_NAME}") String appName, @Value(value = "${health.uri}") String healthUri) {
		this.appName = appName;
		this.healthUri = healthUri;
	}

	@Override
	public Mono<Void> filter(ServerWebExchange exchange, WebFilterChain chain) {
		if (uriSubList.isEmpty()) {
			uriSubList = new HashSet<String>(Arrays.asList(healthUri.split("\\s*/\\s*")));
		}
		String apiPath = exchange.getRequest().getPath() != null ? exchange.getRequest().getPath().toString() : null;
		if (Arrays.asList(apiPath.split("\\s*/\\s*")).containsAll(uriSubList)) {
			return chain.filter(exchange);
		}

		String correlationId = exchange.getRequest().getHeaders() != null && !CollectionUtils.isEmpty(exchange.getRequest().getHeaders().get(AutoPaySynchConstant.CORRELATION_ID))
				? exchange.getRequest().getHeaders().get(AutoPaySynchConstant.CORRELATION_ID).get(0)
				: null;

		log.info("correlationId: {}", correlationId);

		if (ObjectUtils.isEmpty(correlationId)) {
			correlationId = generateUniqueCorrelationId(exchange);
		}

		return chain.filter(exchange).contextWrite(Context.of(AutoPaySynchConstant.CORRELATION_ID, correlationId));
	}

	private String generateUniqueCorrelationId(ServerWebExchange exchange) {
		String[] pathElements = exchange.getRequest().getPath().toString().split("/");

		StringBuilder correlationIdBuffer = new StringBuilder();

		correlationIdBuffer.append(appName).append("_").append(pathElements[pathElements.length - 1]).append("_").append(ZonedDateTime.now());

		return UUIDGenerator.generateType5UUID(correlationIdBuffer.toString()).toString();
	}
}

package com.creditone.ucrm.autopaysynch.kafka.consumer;

import java.time.Duration;
import java.time.Instant;
import java.util.UUID;

import jakarta.annotation.PreDestroy;

import org.springframework.boot.CommandLineRunner;

import org.apache.commons.lang3.exception.ExceptionUtils;

import com.fasterxml.jackson.databind.DeserializationFeature;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;

import com.creditone.microservice.util.UUIDGenerator;
import com.creditone.ucrm.autopaysynch.config.KafkaPropertiesConfig;
import com.creditone.ucrm.autopaysynch.constant.AutoPaySynchConstant;
import com.creditone.ucrm.autopaysynch.constant.AutoPaySynchConstant.SyncType;
import com.creditone.ucrm.autopaysynch.dto.AutoPaySynchLogResponse;

import com.creditone.ucrm.autopaysynch.processor.CommonSyncProcessor;

import io.netty.handler.timeout.TimeoutException;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

import reactor.core.publisher.Flux;
import reactor.core.publisher.Mono;
import reactor.kafka.receiver.ReceiverRecord;
import reactor.util.retry.Retry;

/**
 * Abstract base class for reactive Kafka consumers in the autopay synchronization system.
 * This class provides a standardized framework for consuming Kafka events with built-in resilience
 * patterns, lifecycle management, and error handling.
 *
 * @param <T>
 */
@Slf4j
@RequiredArgsConstructor
public abstract class AbstractKafkaConsumer<T> implements CommandLineRunner {

	protected final KafkaPropertiesConfig properties;
	protected final CommonSyncProcessor commonSyncProcessor;
	protected volatile Disposable consumerDisposable;

	private static final ObjectMapper OBJECT_MAPPER = createObjectMapper();

	private static ObjectMapper createObjectMapper() {
		ObjectMapper mapper = new ObjectMapper();
		mapper.registerModule(new JavaTimeModule());
		mapper.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
		return mapper;
	}

	/**
	 * Initializes and starts the Kafka consumer after bean construction. This method sets up the
	 * reactive consumer stream with comprehensive error handling, retry logic, and lifecycle
	 * callbacks for monitoring and debugging.
	 */
	public void startConsuming() {
		log.info("Starting {} Kafka Consumer", getEventType());

		consumerDisposable = createConsumerFlux().doOnSubscribe(subscription -> log.info("Successfully subscribed to {} events for topic {}", getEventType(), getTopic()))
				.doOnError(error -> log.error("Error consuming {} events for topic {}: {}", getEventType(), getTopic(), ExceptionUtils.getStackTrace(error).replaceAll(
						"[\\r\\n\\t]+", " "), error)).doOnCancel(() -> log.info("{} consumer cancelled for topic {}", getEventType(), getTopic())).retryWhen(this.createRetrySpec())
				.subscribe(event -> log.debug("Successfully processed {} event for topic {}: {}", getEventType(), getTopic(), event), error -> log.error(
						"Fatal error in {} consumer for topic {} : {}", getEventType(), getTopic(), ExceptionUtils.getStackTrace(error).replaceAll("[\\r\\n\\t]+", " "), error),
						() -> log.info("{} consumer stream completed for topic {}", getEventType(), getTopic()));
	}

	/**
	 * Creates a sophisticated retry specification with exponential backoff and jitter. This method
	 * implements enterprise-grade retry logic to handle transient failures gracefully while
	 * preventing system overload.
	 *
	 * @return configured retry specification
	 */
	public Retry createRetrySpec() {
		KafkaPropertiesConfig.Consumer.Common common = properties.getConsumer().getCommon();

		return Retry.backoff(common.getRetryAttempts(), Duration.ofMillis(common.getRetryBackoffMs())).maxBackoff(Duration.ofMillis(common.getRetryMaxDelayMs())).jitter(common
				.getRetryJitter()).filter(this::isRetryableException).doBeforeRetry(retrySignal -> log.warn("Retrying Kafka operation, attempt: {}, error: {}", retrySignal
						.totalRetries() + 1, ExceptionUtils.getStackTrace(retrySignal.failure()).replaceAll("[\\r\\n\\t]+", " "))).onRetryExhaustedThrow(
								(retryBackoffSpec, retrySignal) -> new KafkaRetryExhaustedException("Kafka operation failed after " + common.getRetryAttempts() + " attempts",
										retrySignal.failure()));
	}

	/**
	 * Determines whether an exception should trigger a retry attempt. This method implements
	 * intelligent exception classification to avoid retrying permanent failures while handling
	 * transient issues gracefully.
	 *
	 * @param throwable the exception to evaluate
	 * @return true if the exception is retryable, false otherwise
	 */
	private boolean isRetryableException(Throwable throwable) {
		return throwable instanceof TimeoutException || throwable instanceof org.apache.kafka.common.errors.RetriableException || throwable instanceof java.net.ConnectException
				|| throwable instanceof org.apache.kafka.clients.consumer.CommitFailedException || throwable instanceof org.apache.kafka.common.errors.RebalanceInProgressException
				|| (throwable.getMessage() != null && throwable.getMessage().contains("Offset commit cannot be completed") || throwable instanceof Exception);
	}

	/**
	 * Abstract method that subclasses must implement to create their specific consumer flux. This
	 * method defines the actual Kafka consumption logic including topic subscription, message
	 * deserialization, and event processing pipeline.
	 *
	 * @return Flux of EventWithCorrelation to maintain correlation context
	 */
	protected abstract Flux<EventWithCorrelationData<T>> createConsumerFlux();

	/**
	 * Abstract method that subclasses must implement to identify their event type. This method
	 * provides type identification for logging, monitoring, and debugging purposes.
	 *
	 * @return
	 */
	public abstract SyncType getEventType();

	/**
	 * Abstract method that subclasses must implement to identify the topic. This method
	 * provides topic identification for logging, monitoring, and debugging purposes.
	 *
	 * @return
	 */
	public abstract String getTopic();

	/**
	 * Generates a unique correlation ID for distributed tracing and request correlation. This method
	 * creates deterministic UUIDs based on service name, timestamp, and additional entropy, enabling
	 * unique request tracking per event across multiple services and systems.
	 *
	 * @return unique correlation ID for each event
	 */
	protected String generateUniqueCorrelationId() {
		String name = System.nanoTime() + "_" + UUID.randomUUID();
		return UUIDGenerator.generateType5UUID(name).toString();
	}

	/**
	 * Maps a Kafka consumer record to the target event type with unique correlation ID generation.
	 * This method handles deserialization with per-event correlation tracking.
	 *
	 * @param record           the Kafka consumer record
	 * @param sourceEventClass the target event class
	 * @return mapped event object with correlation context
	 */
	protected EventWithCorrelationData<T> mapRecordWithCorrelation(ReceiverRecord<String, String> record, Class<T> sourceEventClass) {
		String correlationId = generateUniqueCorrelationId();
		Instant startTime = Instant.now();
		try {
			log.info(AutoPaySynchConstant.LOG_PREFIX + "Received {} autopay event at {} - Key: {}, Partition: {}, Offset: {}, Topic: {}, Message: {}", correlationId,
					getEventType(), startTime, record.key(), record.partition(), record.offset(), record.topic(), record.value());
			T event = OBJECT_MAPPER.readValue(record.value(), sourceEventClass);
			return new EventWithCorrelationData<>(event, correlationId, startTime, null, record);
		} catch (Exception e) {
			Throwable error = new AutoPaySynchException("Failed to map message to " + sourceEventClass.getSimpleName() + " for correlationId: " + correlationId + ", record value: "
					+ record.value(), correlationId, e);
			return new EventWithCorrelationData<>(null, correlationId, startTime, error, record);
		}
	}

	/**
	 * Extracts correlation ID from
	 *
	 * @param error the exception that occurred
	 * @return correlation ID or "unknown" if not available
	 */
	protected String extractCorrelationId(Throwable error, Object eventWithCorrelation) {
		if (eventWithCorrelation instanceof EventWithCorrelationData<?> data) {
			return data.getCorrelationId();
		}
		if (error != null) {
			if (error instanceof AutoPaySynchException syncException) {
				if (syncException.getCorrelationId() != null) {
					return syncException.getCorrelationId();
				}
			}
		}
		return "unknown";
	}

	@Override
	public void run(String... args) {
		log.info("Starting {} Kafka consumer for topic: {}", getEventType(), getTopic());
		startConsuming();
	}

	/**
	 * Performs graceful cleanup of consumer resources during application shutdown. This method
	 * ensures proper resource disposal and prevents memory leaks by cleanly terminating the consumer
	 * subscription.
	 */
	@PreDestroy
	public void cleanup() {
		log.info("Shutting down {} Kafka Consumer for topic: {}", getEventType(), getTopic());
		if (consumerDisposable != null && !consumerDisposable.isDisposed()) {
			log.info("Disposing {} consumer subscription for topic: {}", getEventType(), getTopic());
			consumerDisposable.dispose();
			log.info("{} consumer disposed successfully for topic: {}", getEventType(), getTopic());
		}
	}

	/**
	 * Checks the current health status of the Kafka consumer. This method provides a simple health
	 * check mechanism to determine if the consumer is actively processing messages.
	 *
	 * @return true if consumer is active, false otherwise
	 */
	public boolean isConsumerActive() {
		return consumerDisposable != null && !consumerDisposable.isDisposed();
	}

	/**
	 * Handler to update the sync log response with reason based on the error occurred for future reference
	 *
	 * @param eventWithCorrelation
	 * @param logResponse
	 * @param error
	 * @return Mono of ResultDTO
	 */
	protected Mono<ResultDTO> handleError(EventWithCorrelationData<?> eventWithCorrelation, AutoPaySynchLogResponse logResponse, Throwable error) {

		return commonSyncProcessor.handleError(eventWithCorrelation, logResponse, getEventType(), error).doFinally(signalType -> {
			long timeTaken = Duration.between(eventWithCorrelation.getStartTime(), Instant.now()).toMillis();
			log.info(AutoPaySynchConstant.LOG_PREFIX + "[ERROR] Completed processing for {} event for topic: {}, totalTimeTaken: {}. signalType: {} eventMessage: {} error: {}",
					eventWithCorrelation.getCorrelationId(), getEventType(), getTopic(), timeTaken, signalType, eventWithCorrelation.getRecord().value(), error != null
							? ExceptionUtils.getStackTrace(error).replaceAll("[\\r\\n\\t]+", " ")
							: null);
		});
	}

	/**
	 * Log result with failure or success and the time taken to process each event with other details
	 *
	 * @param result
	 * @param eventWithCorrelation
	 */
	protected void logResult(ResultDTO result, EventWithCorrelationData<?> eventWithCorrelation) {

		long timeTaken = Duration.between(eventWithCorrelation.getStartTime(), Instant.now()).toMillis();
		log.info(AutoPaySynchConstant.LOG_PREFIX + "[{}] Completed processing for {} event for topic: {}, totalTimeTaken: {}. eventMessage: {} error: {}", eventWithCorrelation
				.getCorrelationId(), ((result != null && Boolean.TRUE.equals(result.getResult())) ? "SUCCESS" : "FAILURE"), getEventType(), getTopic(), timeTaken,
				eventWithCorrelation.getRecord().value(), (result.getError() != null ? ExceptionUtils.getStackTrace(result.getError()).replaceAll("[\\r\\n\\t]+", " ") : null));
	}

	/**
	 * Save event in database and acknowledge the receiver offset
	 *
	 * @param eventWithCorrelation
	 * @param receiverRecord
	 * @return Mono of AutoPaySynchLogResponse
	 */
	protected Mono<AutoPaySynchLogResponse> saveEventAndAcknowledge(EventWithCorrelationData<?> eventWithCorrelation, ReceiverRecord<String, String> receiverRecord) {

		return commonSyncProcessor.saveEvent(eventWithCorrelation, getEventType()).doOnSuccess(logResponse -> {
			receiverRecord.receiverOffset().acknowledge();
			log.info(AutoPaySynchConstant.LOG_PREFIX + "Successfully saved and acknowledged the event. logResponse: {}", eventWithCorrelation.getCorrelationId(), logResponse);
		});
	}

	/**
	 * Abstract method that subclasses must implement to process their specific event type.
	 * This method defines the actual event processing logic for each consumer type.
	 *
	 * @param event         the deserialized event to process
	 * @param logResponse   the log response from saving the event
	 * @param correlationId unique correlation ID for tracking
	 * @return Mono of ResultDTO containing processing result
	 */
	protected abstract Mono<ResultDTO> processEvent(T event, AutoPaySynchLogResponse logResponse, String correlationId);

	/**
	 * Abstract method that subclasses must implement to get the source event class.
	 * This is needed for deserialization in mapRecordWithCorrelation.
	 *
	 * @return the Class object for the event type
	 */
	protected abstract Class<T> getEventClass();

	/**
	 * Processes a Kafka record with acknowledgment handling.
	 * This method implements the common processing flow for all consumer types:
	 * 1. Map record to event with correlation
	 * 2. Save event to database
	 * 3. Acknowledge Kafka offset
	 * 4. Process the event
	 * 5. Handle any errors
	 *
	 * @param receiverRecord the Kafka receiver record
	 * @return Mono of EventWithCorrelationData
	 */
	protected Mono<EventWithCorrelationData<T>> processWithAck(ReceiverRecord<String, String> receiverRecord) {

		return Mono.fromCallable(() -> mapRecordWithCorrelation(receiverRecord, getEventClass())).flatMap(eventWithCorrelation -> saveEventAndAcknowledge(eventWithCorrelation,
				receiverRecord).flatMap(logResponse -> {
					// Handle mapping errors
					if (eventWithCorrelation.getError() != null) {
						return handleError(eventWithCorrelation, logResponse, eventWithCorrelation.getError());
					}

					// Process the event
					return processEvent(eventWithCorrelation.getEvent(), logResponse, eventWithCorrelation.getCorrelationId()).doOnSuccess(result -> logResult(result,
							eventWithCorrelation)).onErrorResume(error -> handleError(eventWithCorrelation, logResponse, error));
				}).thenReturn(eventWithCorrelation));
	}

}

							error, value), getEventType(), value, ExceptionUtils.getStackTrace(error).replaceAll("[\\r\\n\\t]+", " "));

	 *         legacy-to-modern autopay synchronization events
	 */
	@Override
	public SyncType getEventType() {

	}

	public String getTopic() {

package com.creditone.ucrm.autopaysynch.processor;

import org.apache.commons.lang3.exception.ExceptionUtils;

import com.hazelcast.map.IMap;
import com.hazelcast.nio.serialization.genericrecord.GenericRecord;

import com.creditone.ucrm.autopaysynch.constant.AutoPaySynchConstant;
import com.creditone.ucrm.autopaysynch.constant.AutoPaySynchConstant.SyncType;
import com.creditone.ucrm.autopaysynch.service.ExternalCallService;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import reactor.core.publisher.Mono;

/**

 - Abstract base processor for Hazelcast-based ID mapping with external service fallback. Provides
 - common functionality for converting IDs between modern and legacy systems.
 -
 - @param <T> the return type for converted IDs
 */
@Slf4j
@RequiredArgsConstructor
public abstract class AbstractHazelcastLookupProcessor<T> {

    protected final HazelcastWrapperProcessor hazelcastWrapperProcessor;
    protected final ExternalCallService externalCallService;

    /**
     - Converts an ID based on the specified sync type. First attempts to retrieve from Hazelcast
     - cache, then falls back to external service.
     -
     - @param key           the ID to convert
     - @param syncType      the direction of conversion
     - @param correlationId unique identifier for tracking the request
     - @return Mono containing the converted ID, or empty if conversion fails
     */
    public Mono<T> convertId(String key, SyncType syncType, String correlationId) {
        if (key == null || key.trim().isEmpty()) {
            log.warn(AutoPaySynchConstant.LOG_PREFIX + "Invalid key provided for {} conversion for {}", correlationId, getEntityType(), syncType);
            return Mono.empty();
        }

        log.info(AutoPaySynchConstant.LOG_PREFIX + "{} Converting ID for {}: key={}", correlationId, getEntityType(), syncType, key);

        return getIdFromHazelcast(key, syncType, correlationId).switchIfEmpty(fallbackToExternalService(key, syncType, correlationId)).doOnSuccess(result -> {
            if (result != null) {
                log.info(AutoPaySynchConstant.LOG_PREFIX + "{} ID conversion successful for {}: {} -> {}", correlationId, getEntityType(), syncType, key, result);
            }
        });
    }

    /** Retrieves ID from Hazelcast cache based on sync type configuration. */
    private Mono<T> getIdFromHazelcast(String key, SyncType syncType, String correlationId) {
        try {
            String mapName = getMapName(syncType);
            String keyName = getKeyName(syncType);


            IMap<String, GenericRecord> map = hazelcastWrapperProcessor.getMap(mapName, correlationId);
            if (map == null) {
                log.warn(AutoPaySynchConstant.LOG_PREFIX + "Hazelcast map '{}' not found for {}", correlationId, mapName, syncType);
                return Mono.empty();
            }

            GenericRecord record = map.get(key);
            if (record == null) {
                log.warn(AutoPaySynchConstant.LOG_PREFIX + "Record not found for key '{}' in map '{}' for {}", correlationId, key, mapName, syncType);
                return Mono.empty();
            }

            return processRecord(record, syncType, correlationId, keyName);

        } catch (Exception e) {
            log.warn(AutoPaySynchConstant.LOG_PREFIX + "Error retrieving {} ID for {} from Hazelcast for key={}, error={}, cause={}", getEntityType(), syncType, correlationId,
                    key, e.getMessage(), ExceptionUtils.getStackTrace(e).replaceAll("\\r?\\n", " "));
            return Mono.empty();
        }

    }

    /** Fallback method that delegates to external service when Hazelcast lookup fails. */
    private Mono<T> fallbackToExternalService(String key, SyncType syncType, String correlationId) {
        log.info(AutoPaySynchConstant.LOG_PREFIX + "Falling back to externalCallService for {} for key={}", correlationId, syncType, key);
        return callExternalService(key, syncType, correlationId);
    }

    // Abstract methods to be implemented by concrete classes
    protected abstract String getMapName(SyncType syncType);

    protected abstract String getKeyName(SyncType syncType);

    protected abstract String getEntityType();

    protected abstract Mono<T> processRecord(GenericRecord record, SyncType syncType, String correlationId, String keyName);

    protected abstract Mono<T> callExternalService(String key, SyncType syncType, String correlationId);
}

/**

 - Service responsible for all external HTTP API calls in the autopay synchronization system.
 - Converts legacy IDs to modern IDs (and vice versa) by calling:
 - - Credit Account Service       → getModernCreditAccountId()
 - - CIAM / Customer Domain Service → convertCustomerId()
 - - Agent ID Fallback Service     → convertAgentId()
 */
@Slf4j
@Component
@RefreshScope
@RequiredArgsConstructor
@EnableConfigurationProperties(ExternalCallService.ExternalServiceConfig.class)
public class ExternalCallService {

    private final ExternalServiceConfig externalServiceConfig;
    private final AutoPaySyncClientHandler autoPaySyncClientHandler;


    @Data
    @ConfigurationProperties(prefix = "external-service")
    public static class ExternalServiceConfig {

        private CiamConfig ciam;
        private CreditAccountConfig creditAccount;
        private CoOp hazelcast;

        @Data
        public static class CiamConfig {
            private String host;
            private String productsUri;
            private String apikey;
        }

            private String host;
            private String accountsUri;
            private String apikey;
        }

        @Data
        public static class CoOp {
            private String host;
            private String apikey;
            private String hazelcastAgentIdFallbackServiceUri;
        }

// ─────────────────────────────────────────────────────────────
// getModernCreditAccountId
// Calls: Credit Account Service
// ─────────────────────────────────────────────────────────────

    public Mono<AccountIdDto> getModernCreditAccountId(String modernCustomerId, String correlationId) {
        log.info(AutoPaySynchConstant.LOG_PREFIX + "Request to getModernCreditAccountId() for modernCustomerId: {}",
                correlationId, modernCustomerId);

        var headers = createHeaders(externalServiceConfig.getCreditAccount().getApikey(), correlationId);
        var params = new LinkedMultiValueMap<String, String>();
        params.add("individualId", modernCustomerId);

        String url = externalServiceConfig.getCreditAccount().getHost()
                + externalServiceConfig.getCreditAccount().getAccountsUri();

        return autoPaySyncClientHandler.get(url, headers, "credit-account-service", params)
                .bodyToMono(AccountsDetailsResponse.class)
                .doOnNext(response -> log.info(AutoPaySynchConstant.LOG_PREFIX
                                + "Response from getModernCreditAccountId() for modernCustomerId: {}: {}",
                        correlationId, modernCustomerId, response))
                .flatMap(response -> extractModernCreditAccountId(response, modernCustomerId, correlationId))
                .onErrorResume(err -> {
                    log.error(AutoPaySynchConstant.LOG_PREFIX
                                    + "Error in getModernCreditAccountId() for modernCustomerId: {}, Error: {}",
                            correlationId, modernCustomerId, err.getMessage());
                    return Mono.error(err);
                });
    }

    private Mono<AccountIdDto> extractModernCreditAccountId(AccountsDetailsResponse response,
                                                            String modernCustomerId, String correlationId) {
        if (response == null) {
            return Mono.error(new AutoPaySynchException(
                    "Null response from Credit Account Service for modernCustomerId: " + modernCustomerId,
                    correlationId));
        }

        List<AccountsDetailsResponseContentInner> content = response.getContent();
        if (content == null || content.isEmpty()) {
            return Mono.error(new AutoPaySynchException(
                    "No account data found for modernCustomerId: " + modernCustomerId, correlationId));
        }

        return content.stream()
                .filter(Objects::nonNull)
                .map(AccountsDetailsResponseContentInner::getAccountId)
                .filter(Objects::nonNull)
                .map(Object::toString)
                .filter(StringUtils::hasText)
                .findFirst()
                .map(AccountIdDto::new)
                .map(Mono::just)
                .orElseGet(() -> {
                    log.error(AutoPaySynchConstant.LOG_PREFIX
                                    + "No modern credit account ID found for modernCustomerId: {}",
                            correlationId, modernCustomerId);
                    return Mono.error(new AutoPaySynchException(
                            "No modern credit account ID found for modernCustomerId: " + modernCustomerId,
                            correlationId));
                });
    }

// ─────────────────────────────────────────────────────────────
// convertCustomerId
// Calls: CIAM / Customer Domain Service
// Fallback for: CustomerMappingProcessor
// ─────────────────────────────────────────────────────────────

    public Mono<String> convertCustomerId(String customerId, SyncType syncType, String correlationId) {
        log.info(AutoPaySynchConstant.LOG_PREFIX + "Request to convertCustomerId() for {} for customerId: {}",
                correlationId, syncType, customerId);

        var headers = createHeaders(externalServiceConfig.getCiam().getApikey(), correlationId);
        var params = new LinkedMultiValueMap<String, String>();

        if (SyncType.LEGACY_TO_MODERN.equals(syncType)) {
            params.add("systemCustomerId", customerId);
        } else {
            params.add("individualUniqueIdentifier", customerId);
        }

        String url = externalServiceConfig.getCiam().getHost()
                + externalServiceConfig.getCiam().getProductsUri();

        return autoPaySyncClientHandler.get(url, headers, "customer-domain-service", params)
                .bodyToMono(SystemLinkagesResponse.class)
                .doOnNext(response -> log.info(AutoPaySynchConstant.LOG_PREFIX
                        + "Response from convertCustomerId() for {}: {}", correlationId, syncType, response))
                .flatMap(response -> extractCustomerIdFromResponse(response, syncType, customerId, correlationId))
                .onErrorResume(err -> {
                    log.error(AutoPaySynchConstant.LOG_PREFIX
                                    + "Error in convertCustomerId() for {}, Error: {}",
                            correlationId, syncType, err.getMessage());
                    return Mono.error(err);
                });
    }

    private Mono<String> extractCustomerIdFromResponse(SystemLinkagesResponse response, SyncType syncType,
                                                       String customerId, String correlationId) {
        if (response == null) {
            return Mono.error(new AutoPaySynchException(
                    "Null response from CIAM service for customerId: " + customerId, correlationId));
        }

        List<SystemLinkage> content = response.getContent();
        if (content == null || content.isEmpty()) {
            return Mono.error(new AutoPaySynchException(
                    "No system linkage data found for customerId: " + customerId + " syncType: " + syncType,
                    correlationId));
        }

        return switch (syncType) {
            case LEGACY_TO_MODERN -> extractModernCustomerId(content, customerId, correlationId);
            case MODERN_TO_LEGACY -> extractLegacyCustomerId(content, customerId, correlationId);
        };
    }

    private Mono<String> extractModernCustomerId(List<SystemLinkage> content, String customerId,
                                                 String correlationId) {
        return content.stream()
                .filter(Objects::nonNull)
                .map(SystemLinkage::getIndividualUniqueIdentifier)
                .filter(Objects::nonNull)
                .map(Object::toString)
                .filter(StringUtils::hasText)
                .findFirst()
                .map(Mono::just)
                .orElseGet(() -> {
                    log.error(AutoPaySynchConstant.LOG_PREFIX
                                    + "No modern customer ID found for legacyCustomerId: {}",
                            correlationId, customerId);
                    return Mono.error(new AutoPaySynchException(
                            "No modern customer ID found for legacyCustomerId: " + customerId, correlationId));
                });
    }

    private Mono<String> extractLegacyCustomerId(List<SystemLinkage> content, String customerId,
                                                 String correlationId) {
        return content.stream()
                .filter(Objects::nonNull)
                .map(SystemLinkage::getSystemCustomerId)
                .filter(StringUtils::hasText)
                .findFirst()
                .map(Mono::just)
                .orElseGet(() -> {
                    log.error(AutoPaySynchConstant.LOG_PREFIX
                                    + "No legacy customer ID found for modernCustomerId: {}",
                            correlationId, customerId);
                    return Mono.error(new AutoPaySynchException(
                            "No legacy customer ID found for modernCustomerId: " + customerId, correlationId));
                });
    }

// ─────────────────────────────────────────────────────────────
// convertAgentId
// Calls: Agent ID Fallback Service
// Fallback for: AgentMappingProcessor
// ─────────────────────────────────────────────────────────────

    public Mono<String> convertAgentId(String agentId, SyncType syncType, String correlationId) {
        log.info(AutoPaySynchConstant.LOG_PREFIX + "Request to convertAgentId() for {} for agentId: {}",
                correlationId, syncType, agentId);

        var headers = new HashMap<String, String>();
        headers.put(AutoPaySynchConstant.CORRELATION_ID, correlationId);
        headers.put("apiKey", externalServiceConfig.getHazelcast().getApikey());
        headers.put("requestedBy", AutoPaySynchConstant.AUTOPAY_SYNCH_SERVICE);

        var params = new LinkedMultiValueMap<String, String>();
        if (SyncType.LEGACY_TO_MODERN.equals(syncType)) {
            params.add("casId", agentId);
        } else {
            params.add("userUniqueIdentifier", agentId);
        }

        String url = externalServiceConfig.getHazelcast().getHost()
                + externalServiceConfig.getHazelcast().getHazelcastAgentIdFallbackServiceUri();

        return autoPaySyncClientHandler.get(url, headers, "agent-id-fallback-service", params)
                .bodyToMono(PageUserResponse.class)
                .doOnNext(response -> log.info(AutoPaySynchConstant.LOG_PREFIX
                        + "Response from convertAgentId() for {}: {}", correlationId, syncType, response))
                .flatMap(response -> extractAgentIdFromResponse(response, syncType, agentId, correlationId))
                .onErrorResume(err -> {
                    log.error(AutoPaySynchConstant.LOG_PREFIX
                                    + "Error in convertAgentId() for {}, Error: {}",
                            correlationId, syncType, err.getMessage());
                    return Mono.error(err);
                });
    }

    private Mono<String> extractAgentIdFromResponse(PageUserResponse response, SyncType syncType,
                                                    String agentId, String correlationId) {
        if (response == null) {
            return Mono.error(new AutoPaySynchException(
                    "Null response from Agent ID Fallback Service for agentId: " + agentId, correlationId));
        }

        List<UserResponse> content = response.getContent();
        if (content == null || content.isEmpty()) {
            return Mono.error(new AutoPaySynchException(
                    "No user data found for agentId: " + agentId + " syncType: " + syncType, correlationId));
        }

        return switch (syncType) {
            case LEGACY_TO_MODERN -> extractModernAgentId(content, agentId, correlationId);
            case MODERN_TO_LEGACY -> extractLegacyAgentId(content, agentId, correlationId);
        };
    }

    private Mono<String> extractModernAgentId(List<UserResponse> content, String agentId, String correlationId) {
        return content.stream()
                .filter(Objects::nonNull)
                .map(UserResponse::getUniqueIdentifier)
                .filter(Objects::nonNull)
                .map(Object::toString)
                .filter(StringUtils::hasText)
                .findFirst()
                .map(Mono::just)
                .orElseGet(() -> {
                    log.error(AutoPaySynchConstant.LOG_PREFIX
                            + "No modern agent ID found for legacy casId: {}", correlationId, agentId);
                    return Mono.error(new AutoPaySynchException(
                            "No modern agent ID found for legacy casId: " + agentId, correlationId));
                });
    }

    private Mono<String> extractLegacyAgentId(List<UserResponse> content, String agentId, String correlationId) {
        return content.stream()
                .filter(Objects::nonNull)
                .map(UserResponse::getMetaData)
                .filter(Objects::nonNull)
                .map(UserResponseMetaData::getCasId)
                .filter(StringUtils::hasText)
                .findFirst()
                .map(Mono::just)
                .orElseGet(() -> {
                    log.error(AutoPaySynchConstant.LOG_PREFIX
                            + "No legacy casId found for modern agentId: {}", correlationId, agentId);
                    return Mono.error(new AutoPaySynchException(
                            "No legacy casId found for modern agentId: " + agentId, correlationId));
                });
    }

// ─────────────────────────────────────────────────────────────

    private Map<String, String> createHeaders(String apiKey, String correlationId) {
        return Map.of("apiKey", apiKey, AutoPaySynchConstant.CORRELATION_ID, correlationId);
    }
