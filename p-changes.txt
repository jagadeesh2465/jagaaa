package com.creditone.ucrm.autopaysynch.config;

import java.time.Duration;

import jakarta.annotation.PreDestroy;

import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.boot.context.properties.EnableConfigurationProperties;
import org.springframework.cloud.context.config.annotation.RefreshScope;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.data.r2dbc.core.DefaultReactiveDataAccessStrategy;
import org.springframework.data.r2dbc.core.R2dbcEntityOperations;
import org.springframework.data.r2dbc.core.R2dbcEntityTemplate;
import org.springframework.data.r2dbc.dialect.PostgresDialect;
import org.springframework.data.r2dbc.repository.config.EnableR2dbcRepositories;
import org.springframework.r2dbc.connection.R2dbcTransactionManager;
import org.springframework.r2dbc.core.DatabaseClient;
import org.springframework.transaction.ReactiveTransactionManager;

import org.apache.commons.lang3.exception.ExceptionUtils;

import io.r2dbc.pool.ConnectionPool;
import io.r2dbc.pool.ConnectionPoolConfiguration;
import io.r2dbc.spi.ConnectionFactories;
import io.r2dbc.spi.ConnectionFactory;
import io.r2dbc.spi.ConnectionFactoryOptions;
import lombok.Data;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;

import static io.r2dbc.spi.ConnectionFactoryOptions.DATABASE;
import static io.r2dbc.spi.ConnectionFactoryOptions.DRIVER;
import static io.r2dbc.spi.ConnectionFactoryOptions.HOST;
import static io.r2dbc.spi.ConnectionFactoryOptions.PASSWORD;
import static io.r2dbc.spi.ConnectionFactoryOptions.PORT;
import static io.r2dbc.spi.ConnectionFactoryOptions.USER;

@Slf4j
@Configuration
@RefreshScope
@EnableConfigurationProperties(AutoPayDBConfig.DatabaseConfig.class)
@EnableR2dbcRepositories(entityOperationsRef = "autopayEntityTemplate", basePackages = "${database.autopay.repository-base-package}")
@RequiredArgsConstructor
public class AutoPayDBConfig {

	private final DatabaseConfig databaseConfig;
	private ConnectionPool connectionPool;

	@Data
	@ConfigurationProperties("database.autopay")
	public static class DatabaseConfig {
		private String host;
		private Integer port;
		private String username;
		private String password;
		private String databaseName;
		private Integer initialPoolSize;
		private Integer maxPoolSize;
		private Integer maxIdleTime;
		private String repositoryBasePackage;
		private Integer maxAcquireTime;
		private Integer maxCreateConnectionTime;
		private Integer maxLifeTime;
		private Integer acquireRetry;
	}

	@Bean
	@Qualifier(value = "autopayConnectionFactory")
	public ConnectionFactory autopayConnectionFactory() {
		log.info("Initializing AutoPay database connection factory for host: {}, database: {}", databaseConfig.getHost(), databaseConfig.getDatabaseName());

		try {
			ConnectionFactoryOptions options = ConnectionFactoryOptions.builder().option(DRIVER, "postgresql").option(HOST, databaseConfig.getHost()).option(PORT, databaseConfig
					.getPort()).option(DATABASE, databaseConfig.getDatabaseName()).option(USER, databaseConfig.getUsername()).option(PASSWORD, databaseConfig.getPassword())
					.build();
			ConnectionFactory connectionFactory = ConnectionFactories.get(options);
			ConnectionPoolConfiguration poolConfiguration = ConnectionPoolConfiguration.builder(connectionFactory).initialSize(databaseConfig.getInitialPoolSize()).maxSize(
					databaseConfig.getMaxPoolSize()).maxIdleTime(Duration.ofMinutes(databaseConfig.getMaxIdleTime())).maxAcquireTime(Duration.ofSeconds(databaseConfig
							.getMaxAcquireTime())).maxCreateConnectionTime(Duration.ofSeconds(databaseConfig.getMaxCreateConnectionTime())).maxLifeTime(Duration.ofMinutes(
									databaseConfig.getMaxLifeTime())).acquireRetry(databaseConfig.getAcquireRetry()).validationQuery("SELECT 1").build();
			connectionPool = new ConnectionPool(poolConfiguration);

			log.info("AutoPay database connection pool initialized successfully");

			return connectionPool;
		} catch (Exception e) {
			log.error("Failed to initialize AutoPay database connection factory Error: {} Cause: {}", e.getMessage(), ExceptionUtils.getStackTrace(e).replaceAll("\\r?\\n", " "));
			throw new IllegalStateException("Database connection initialization failed", e);
		}
	}

	@Bean
	public R2dbcEntityOperations autopayEntityTemplate(@Qualifier("autopayConnectionFactory") ConnectionFactory connectionFactory) {
		log.info("Creating AutoPay R2DBC entity template");
		DefaultReactiveDataAccessStrategy strategy = new DefaultReactiveDataAccessStrategy(PostgresDialect.INSTANCE);
		DatabaseClient databaseClient = DatabaseClient.builder().connectionFactory(connectionFactory).bindMarkers(PostgresDialect.INSTANCE.getBindMarkersFactory()).build();
		return new R2dbcEntityTemplate(databaseClient, strategy);
	}

	@Bean
	@Qualifier("autopayTransactionManager")
	public ReactiveTransactionManager autopayTransactionManager(@Qualifier("autopayConnectionFactory") ConnectionFactory connectionFactory) {

		log.info("Creating AutoPay transaction manager");
		return new R2dbcTransactionManager(connectionFactory);
	}

	/** Graceful shutdown of connection pool */
	@PreDestroy
	public void cleanup() {
		log.info("Shutting down AutoPay database connection pool");
		try {
			if (connectionPool != null && !connectionPool.isDisposed()) {
				connectionPool.dispose();
				log.info("AutoPay database connection pool disposed successfully");
			}
		} catch (Exception e) {
			log.error("Error during AutoPay database connection pool cleanup Error: {}, Cause: {}", e.getMessage(), ExceptionUtils.getStackTrace(e).replaceAll("\\r?\\n", " "));
		}
	}

}
